  :root{
    --left-rot: -8deg;   /* default left rotation */

  }

  /* use the tokens when positioning left/right cards */
  .card[data-pos="left"]{

  }
  .card[data-pos="right"]{
    left: calc(50% + 380px);
    transform: translateX(60px) rotate(var(--right-rot)) scale(.88);
  }

  /* smoother transitions when angle changes */
  .card {
    transition: transform 420ms cubic-bezier(.2,.9,.2,1), box-shadow 240ms, z-index 0ms;
  }

  /* small control UI */
  .rotate-controls {
    position:absolute;
    right:18px;

<!-- ----- ADD / MERGE INTO <script> (after DOM loaded) ----- -->
<script>
  // wire up controls to CSS variables
  const root = document.documentElement;
  const leftRange = document.getElementById('leftRange');
  const rightRange = document.getElementById('rightRange');
  const leftVal = document.getElementById('leftVal');
  const rightVal = document.getElementById('rightVal');
  const presets = document.querySelectorAll('.preset-btn');

  function setLeft(deg){
    root.style.setProperty('--left-rot', deg + 'deg');
    leftVal.textContent = deg;
    leftRange.value = deg;
  }
  function setRight(deg){
    root.style.setProperty('--right-rot', deg + 'deg');
    rightVal.textContent = deg;
    rightRange.value = deg;
  }

  leftRange.addEventListener('input', (e)=> setLeft(e.target.value));
  rightRange.addEventListener('input', (e)=> setRight(e.target.value));

  presets.forEach(btn => {
    btn.addEventListener('click', () => {
      const l = parseFloat(btn.dataset.left);
      const r = parseFloat(btn.dataset.right);
      setLeft(l);
      setRight(r);
    });
  });

  // initialize (in case defaults differ)
  setLeft(parseFloat(leftRange.value));
  setRight(parseFloat(rightRange.value));
</script>
